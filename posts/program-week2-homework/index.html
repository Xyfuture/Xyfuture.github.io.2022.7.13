int C=0;

{
    int a;
    int b;

struct vis//è®¿é—®æƒ…å†µ

    int a;int b;int v;

situ changeSitu (const situ& cur,int i)//æ”¹å˜çŠ¶æ€ä¸€å…±6æ¡è·¯
{
    situ ns={cur.a,cur.b};

        ns.a = 0;
    else if (i==2)
        ns.a = maxA;
    else if (i==3)
        ns.b=0;
