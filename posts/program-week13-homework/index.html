    A - TT 的神秘任务1（必做）#

    题面#

    解法#

    这道题考察了一些基本的数学知识.

    首先核心就是 奇数+奇数=偶数 偶数+偶数=奇数 奇数+偶数=奇数

    然后还一个就是经常忘的情况,0是偶数,(虽然这个题要求从1开始)

    解题思路就是偶数情况,前k-1个数为2,最后一个数(n-2k+2)要求是偶数

    奇数情况前k-1个数是1,最后一个数(n-k+1)要求是奇数

    如果这两种情况都不能满足那么说明不能满足

    code#

    #include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;

int data;
int k;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&data,&k);
        if((data-2*k+2)>0&&(data-2*k+2)%2==0)//偶数情况,需要判断大于0
        {
            printf("YES\n");
            for(int t=0;t<k-1;t++)//先输出2
                printf("2 ");
            printf("%d\n",(data-2*k+2));//最后一个偶数
        }
        else if((data-k+1)>0&&(data-k+1)%2==1)//奇数情况
        {
            printf("YES\n");
            for(int t=0;t<k-1;t++)//先输出1
                printf("1 ");
            printf("%d\n",data-k+1);//最后一个奇数
        }
        else//不能满足
            printf("NO\n");
    }
}

    总结#

    奇偶判断，感觉比较简单

    B - TT 的神秘任务2（必做）#

    题面#

    解法#

    这道题采用二分的方法,复杂度O(log n)

    二分的方法与之前课上的例题类似,二分查找第一个满足无法整除次数的数.

    这道题的两个数据范围都1e9开始,但是对于二分而言,最大值肯定是超了int了,因此使用long long

    code#

    #include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <climits>
using namespace std;

long long n,k;

bool findCount(long long num)//计算出num是第几个无法被n整除的数,同时比较其与指定个数的大小
{
    long long tt = num-(num/n);
    return tt>=k;
}

int main()
{
    int t;
    scanf("%d",&t);
    for(int m=0;m<t;m++)
    {
        scanf("%lld%lld",&n,&k);
        if(n>k)
        {
            printf("%d\n",k);
            continue;
        }
        long long l=1;//会超int
        long long r=LLONG_MAX;
        long long mid = (l+r)/2;
        long long ans = 0;
        while(r>=l)//二分查找
        {
            if(findCount(mid))//向左
            {
                ans=mid;
                r=mid-1;
            }
            else //向右
                l=mid+1;
            mid = (l+r)/2;
        }
        printf("%lld\n",ans);
    }
}

    总结#

    重新复习一下二分查找

    C - TT 的奖励（必做）#

    题面#

    解法#

    这道题采用dp的方法求解,复杂度大约为O(m)

    我们dp状态设为二维,dp[i][j]表示第i秒再第j个位置上能获得最大猫咪数

    初始状态dp[0][0..10] = 0.

    状态转移分为两种状态,第一个是在初始几秒,移动范围受限,这时只更新能够移动到的范围,第二种就是全局都能移动到

    dp[i][j] = max(dp[i-1][j],dp[i-1][j-1],dp[i-1][j+1])+data[i][j]

    同时处理好边界的关系,在0处和10处进行特判

    code#

    #include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>

using namespace std;

int ans[100005][15];//存储dp结果
int data[100005][15];//存储每一秒当前位置上的猫数

int main()
{
    int m;
    while(scanf("%d",&m)!=EOF)
    {
        if(m==0)
            return 0;
        memset(ans,0,sizeof(ans));
        memset(data,0,sizeof(data));
        int t=0;
        for(int i=0;i<m;i++)
        {
            int a,b;
            scanf("%d%d",&a,&b);
            data[b][a]++;//计数
            if(b>t)//最大秒数
                t=b;
        }
        for(int i=1;i<=t;i++)
        {
            int l = 5-t<0?0:5-t;//主要是因为一开始限制边界的范围
            int r = 5+t>10?10:5+t;
            for(int j=l;j<=r;j++)
            {
                if(j==0)//在0处和10处需要特判
                    ans[i][j] = max(ans[i-1][0],ans[i-1][1])+data[i][j];
                else if(j==10)
                    ans[i][j] = max(ans[i-1][9],ans[i-1][10])+data[i][j];
                else
                    ans[i][j] = max(max(ans[i-1][j],ans[i-1][j-1]),ans[i-1][j+1])+data[i][j];//转移方程
            }
        }
        int cur_max=0;
        for(int i=0;i<=10;i++)//取最后一秒最大的值
            if(ans[t][i]>cur_max)
                cur_max = ans[t][i];
        printf("%d\n",cur_max);
    }
    return 0;
}

    总结#

    注意一开始的移动范围，只能从中间开始。
